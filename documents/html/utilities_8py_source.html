107 def normalizeValueString(self, string, value_type):
108 '''
109 @brief Normalize a value string.
110 @param string: The string to normalize.
111 @param value_type: The type of the value.
112 @return: The normalized string.
113 '''
114 supported_types = {"integer", "float", "vector2", "vector3", "vector4", "color3", "color4", "matrix33", "matrix44"}
115
116 if value_type not in supported_types:
117 return string
118
119 token_separator = ", "
120 result = ""
121
122 is_integer = (value_type == "integer")
123
124 if is_integer:
125 # Remove leading and trailing spaces
126 result = string.strip()
127 else:
128 tokens = string.split(',')
129 for token in tokens:
130 # Remove leading and trailing spaces
131 token = token.strip()
132
133 # Skip if the token is empty
134 if not token:
135 continue
136
137 # Remove leading zeros
138 token = token.lstrip('0')
139
140 # Preserve 0 values
141 if token == "" or token[0] == '.':
142 token = "0" + token
143
144 # Check if token has 'e' character for scientific notation
145 if 'e' not in token:
146 # If token has a decimal point, remove trailing zeros
147 if '.' in token:
148 token = token.rstrip('0')
149 # If token ends with '.', remove it
150 if token[-1] == '.':
151 token = token[:-1]
152
153 # Append the formatted token to result with separator
154 result += token + token_separator
155
156 # Remove the last separator
157 if result:
158 result = result[:-len(token_separator)]
159
160 return result

162 def normalizeValueStrings(self, doc):
163 '''
164 @brief Normalize all value strings in a document.
165 @param doc: The document to normalize.
166 '''
167 for elem in doc.traverseTree():
168 if not elem.isA(mx.ValueElement):
169 continue
170
171 original_string = elem.getValueString()
172 if not original_string:
173 continue
174
175 normalized_string = self.normalizeValueString(original_string, elem.getType())
176 if normalized_string != original_string:
177 elem.setValueString(normalized_string)

179 def functionallyEquivalent(self, lhs, rhs):
180 '''
181 @brief Check if two MaterialX elements are functionally equivalent.
182 @param lhs: The first element.
183 @param rhs: The second element.
184 @return: True if the elements are functionally equivalent, otherwise False.'''
185
186 if lhs.getCategory() != rhs.getCategory():
187 self.errorerror = "Mismatched category: " + lhs.getName() + " vs " + rhs.getName()
188 return False
189
190 if lhs.getName() != rhs.getName():
191 self.errorerror = "Mismatched name: " + lhs.getName() + " vs " + rhs.getName()
192 return False
193
194 # Compare attributes.
195 if (len(lhs.getAttributeNames()) != len(rhs.getAttributeNames())):
196 self.errorerror = "Mismatched number of attributes: " , len(lhs.getAttributeNames()) , " vs " , len(rhs.getAttributeNames())
197 return False
198
199 lhsAttrNames = lhs.getAttributeNames()
200 rhsAttrNames = rhs.getAttributeNames()
201 lhsAttrNames.sort()
202 rhsAttrNames.sort()
203 if (lhsAttrNames != rhsAttrNames):
204 self.errorerror = "Mismatched attribute names"
205 return False
206
207 for attr in lhsAttrNames:
208 if lhs.getAttribute(attr) != rhs.getAttribute(attr):
209 self.errorerror = "Mismatched attribute: " + lhs.getAttribute(attr) + " vs " + rhs.getAttribute(attr)
210 return False
211
212 # Compare children.
213 c1 = lhs.getChildren()
214 c2 = rhs.getChildren()
215 if len(c1) != len(c2):
216 self.errorerror = "Mismatched number of children: " , len(c1) , " vs " , len(c2)
217 return False
218
219 # Compare children names
220 c1names = []
221 for child in c1:
222 c1names.append(child.getName())
223 c2names = []
224 for child in c2:
225 c2names.append(child.getName())
226 c1names.sort()
227 c2names.sort()
228 if c1names != c2names:
229 self.errorerror = "Mismatched children names"
230 return False
231
232 for child in c1:
233
234 child2 = rhs.getChild(child.getName())
235
236 #if (child2 is None):
237 # print("Missing child: " + child.getName())
238 # return False
239
240 if not self.functionallyEquivalent(child, child2):
241 return False
242
243 return True
