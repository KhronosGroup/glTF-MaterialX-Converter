70 def stringToScalar(self, value, type):
71 """
72 Convert a supported MaterialX value string to a JSON scalar value.
73 @param value: The string value to convert.
74 @param type: The type of the value.
75 @return The converted scalar value if successful, otherwise the original string value.
76 """
77 return_value = value
78
79 scalar_types = ['integer', 'matrix33', 'matrix44', 'vector2', 'vector3', 'vector4', 'float', 'color3', 'color4']
80
81 if type in scalar_types:
82 split_value = value.split(',')
83
84 if len(split_value) > 1:
85 return_value = list(map(float, split_value))
86 else:
87 if type == 'integer':
88 return_value = int(value)
89 else:
90 return_value = float(value)
91
92 return return_value

363 def materialXtoGLTF(self, mtlxDoc):
364 """
365 @brief Convert a MaterialX document to glTF.
366 @param mtlxDoc: The MaterialX document to convert.
367 @return glTF JSON string and status message.
368 """
369
370 status = ''
371 if not mtlxDoc:
372 status = 'Invalid document to convert'
373 return None, status
374
375 materials = []
376 mxMaterials = mtlxDoc.getMaterialNodes()
377 if len(mxMaterials) == 0:
378 self.logger.warning('> No materials found in document')
379 #return None, status # No MaterialX materials found in the document
380
381 json_data = {}
382 json_asset = {
383 "version": "2.0",
384 "generator": "MaterialX 1.39 / glTF 2.0 procedural texture converter",
385 }
386
387 inputMaps = {}
388 # Supported input mappings
389 inputMaps[MTLX_GLTF_PBR_CATEGORY] = [
390 ['base_color', 'baseColorTexture', 'pbrMetallicRoughness']
391 ]
392
393 pbrNodes = {}
394 fallbackTextureIndex = -1
395 procs = []
396 exportGraphNames = []
397
398 extensionsUsed = ['KHR_texture_procedurals', 'EXT_texture_procedurals_mx_1_39']
399
400 for mxMaterial in mxMaterials:
401 mxshaders = mx.getShaderNodes(mxMaterial)
402
403 # Scan for shaders for the material
404 for shaderNode in mxshaders:
405 category = shaderNode.getCategory()
406 path = shaderNode.getNamePath()
407 isPBR = (category == MTLX_GLTF_PBR_CATEGORY)
408 if (isPBR) and pbrNodes.get(path) is None:
409 self.logger.info(f'> Convert shader to glTF: {shaderNode.getNamePath()}. Category: {category}')
410 pbrNodes[path] = shaderNode
411
412 material = {}
413
414 material['name'] = path
415
416 base_color_input = None
417 base_color_output = ''
418 inputPairs = inputMaps[category]
419
420 # Scan through support inupt channels
421 for inputPair in inputPairs:
422 base_color_input = shaderNode.getInput(inputPair[0])
423 base_color_output = inputPair[1]
424
425 if not base_color_input:
426 continue
427
428 # Check for upstream nodegraph connection. Skip if not found
429 nodeGraphName = base_color_input.getNodeGraphString()
430 if len(nodeGraphName) == 0:
431 continue
432
433 # Check for upstream nodegraph output connection.
434 nodeGraphOutput = base_color_input.getOutputString()
435
436 # Determine the parent of the input
437 parent = material
438 if len(inputPair[2]) > 0:
439 if inputPair[2] not in material:
440 material[inputPair[2]] = {}
441 parent = material[inputPair[2]]
442
443 # Check for an existing converted graph and / or output index
444 # in the "procedurals" list
445 graphIndex = -1
446 outputIndex = -1
447 if procs:
448 for i, proc in enumerate(procs):
449 if proc['name'] == nodeGraphName:
450 graphIndex = i
451 if len(nodeGraphOutput) > 0:
452 for j, output in enumerate(proc['outputs']):
453 if output['name'] == nodeGraphOutput:
454 outputIndex = j
455 break
456 break
457
458 # Make the connection to the input on the material if the graph is already converted
459 if graphIndex >= 0:
460 baseColorTexture = parent[base_color_output] = {}
461 baseColorTexture['index'] = fallbackTextureIndex
462 ext = baseColorTexture['extensions'] = {}
463 lookup = ext['KHR_texture_procedurals'] = {}
464 lookup['index'] = graphIndex
465 if outputIndex >= 0:
466 lookup['output'] = outputIndex
467 else:
468 graph = mtlxDoc.getNodeGraph(nodeGraphName)
469 exportGraphNames.append(nodeGraphName)
470
471 gltfInfo = self.materialXGraphToGLTF(graph, json_data, materials)
472 procs = gltfInfo[0]
473 outputNodes = gltfInfo[1]
474 fallbackTextureIndex = gltfInfo[3]
475
476 baseColorTexture = parent[base_color_output] = {}
477 baseColorTexture['index'] = fallbackTextureIndex
478 ext = baseColorTexture['extensions'] = {}
479 lookup = ext['KHR_texture_procedurals'] = {}
480 lookup['index'] = len(procs) - 1
481 outputIndex = -1
482
483 # Assign an output index if the graph has more than one output
484 if len(nodeGraphOutput) > 0:
485 nodeGraphOutputPath = f"{nodeGraphName}/{nodeGraphOutput}"
486 if nodeGraphOutputPath in outputNodes:
487 outputIndex = outputNodes[nodeGraphOutputPath]
488 else:
489 self.logger.error(f'> Failed to find output: {nodeGraphOutput} '
490 ' in: {outputNodes}')
491 lookup['output'] = outputIndex
492 else:
493 lookup['output'] = 0
494
495 if 'name' in material:
496 materials.append(material)
497
498 unconnectedGraphs = []
499 for ng in mtlxDoc.getNodeGraphs():
500 ng_name = ng.getName()
501 if ng.getAttribute('nodedef') or ng.hasSourceUri():
502 continue
503 if ng_name not in exportGraphNames:
504 unconnectedGraphs.append(ng_name)
505 gltfInfo = self.materialXGraphToGLTF(ng, json_data, materials)
506 procs = gltfInfo[0]
507 outputNodes = gltfInfo[1]
508 fallbackTextureIndex = gltfInfo[3]
509
510 if len(materials) > 0:
511 json_data['materials'] = materials
512 if len(unconnectedGraphs) > 0:
513 status = 'Exported unconnected graphs: ' + ', '.join(unconnectedGraphs)
514 else:
515 if len(unconnectedGraphs) > 0:
516 status = 'Exported unconnected graphs: ' + ', '.join(unconnectedGraphs)
517 else:
518 status = 'No appropriate glTF shader graphs found'
519
520 if len(procs) > 0:
521 json_data['asset'] = json_asset
522 json_data['extensionsUsed'] = extensionsUsed
523
524 jsonString = json.dumps(json_data, indent=2) if json_data else ''
525 if jsonString == '{}':
526 jsonString = ''
527
528 return jsonString, status
