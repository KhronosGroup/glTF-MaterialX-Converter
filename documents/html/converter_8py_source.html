182 def stringToScalar(self, value, type):
183 """
184 Convert a supported MaterialX value string to a JSON scalar value.
185 @param value: The string value to convert.
186 @param type: The type of the value.
187 @return The converted scalar value if successful, otherwise the original string value.
188 """
189 return_value = value
190
191 scalar_types = ['integer', 'matrix33', 'matrix44', 'vector2', 'vector3', 'vector4', 'float', 'color3', 'color4']
192
193 if type in scalar_types:
194 split_value = value.split(',')
195
196 if len(split_value) > 1:
197 return_value = list(map(float, split_value))
198 else:
199 if type == 'integer':
200 return_value = int(value)
201 else:
202 return_value = float(value)
203
204 return return_value

535 continue
